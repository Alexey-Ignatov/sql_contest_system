SELECT customer_rk as CUSTOMER_RK,
(
	CASE
		WHEN RIGHT(middle_nm, 1) = 'Ч' THEN 'M'
		WHEN RIGHT(middle_nm, 1) = 'А' THEN 'F'
		WHEN middle_nm is NULL AND RIGHT(last_nm, 1) = "В" THEN 'M'
		WHEN middle_nm is NULL AND RIGHT(last_nm, 1) = "А" THEN 'F'
        ELSE NULL
	END
) as GENDER
FROM srcdt.cd_customers
where YEAR(valid_to_dttm) = 5999
