select substring(birth_dt, 1, 4) as birth_year, count(distinct customer_rk) as num_distinct
from srcdt.cd_customers
where customer_rk in
(
	select customer_rk
	from srcdt.cd_customers
	where monthly_income_amt = 50000
)
and customer_rk in
(
	select customer_rk
    from srcdt.cd_customers
    where monthly_income_amt = 60000
)
and birth_dt is not null
group by birth_year
