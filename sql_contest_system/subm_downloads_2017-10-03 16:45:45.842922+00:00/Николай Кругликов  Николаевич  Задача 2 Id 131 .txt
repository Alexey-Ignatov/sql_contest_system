SELECT 
    year(birth_dt) as birth_year, count(distinct customer_rk) as cnt
FROM
    srcdt.cd_customers
WHERE
    customer_rk IN (SELECT DISTINCT
            customer_rk
        FROM
            srcdt.cd_customers
        WHERE
            monthly_income_amt = 50000)
        AND customer_rk IN (SELECT DISTINCT
            customer_rk
        FROM
            srcdt.cd_customers
        WHERE
            monthly_income_amt = 60000)
group by birth_year
