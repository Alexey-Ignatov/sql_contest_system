select y as birth_year, count(1) as cnt
from 
(
select distinct customer_rk, year(birth_dt) as y
from srcdt.cd_customers 
where customer_rk in (
select distinct  customer_rk
from srcdt.cd_customers
where monthly_income_amt=50000
)
and customer_rk in (select distinct customer_rk
from srcdt.cd_customers 
where monthly_income_amt=60000)
) as a
group by y;
