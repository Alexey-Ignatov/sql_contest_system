select year(birth_dt) as birth_dt, count(*) as cnt from
(
select distinct birth_dt, customer_rk from srcdt.cd_customers
where customer_rk in (
select distinct customer_rk from srcdt.cd_customers
where monthly_income_amt = 50000
) and customer_rk in (
select distinct customer_rk from srcdt.cd_customers
where monthly_income_amt = 60000
)
) as t
group by year(birth_dt);
