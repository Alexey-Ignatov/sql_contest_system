use srcdt;

select 
	monthly_income_amt
from cd_customers
where
		valid_from_dttm is not null
	and
		valid_to_dttm is not null
	and
		monthly_income_amt is not null
	and
( 
		SUBSTRING(valid_from_dttm, 1, 4) <= '2014'
	and
		SUBSTRING(valid_to_dttm, 1, 4) >= '2014'
 )
order by 
	monthly_income_amt desc
limit 10;
